# Factor 5: Build, release, run
FROM node:14 as build

WORKDIR /app

COPY package*.json ./
RUN npm ci

COPY . .
RUN npm run build

FROM node:14-slim

WORKDIR /app

COPY --from=build /app/dist ./dist
COPY package*.json ./
COPY config ./config
RUN npm ci --only=production

# Factor 3: Config
ENV NODE_ENV=production

# Factor 7: Port binding
EXPOSE 3000

CMD ["node", "dist/index.js"]
